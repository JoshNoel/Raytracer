#Dir: Root
cmake_minimum_required (VERSION 2.6)
project (Raytracer)
SET(Raytracer_VERSION_MAJOR 1)
SET(Raytracer_VERSION_MINOR 0)

ADD_SUBDIRECTORY(src)

FIND_PATH(GLM_INCLUDE_DIR "${PROJECT_SOURCE_DIR}/lib/glm")
INCLUDE_DIRECTORIES(${GLM_INCLUDE_DIR})

FIND_PATH(LODEPNG_DIR "")
INCLUDE_DIRECTORIES(${LODEPNG_DIR})
ADD_LIBRARY(LodePNG ${LODEPNG_DIR}/lodepng.cpp)
TARGET_LINK_LIBRARIES(Raytracer LodePNG)
INSTALL(TARGETS LodePNG DESTINATION "${PROJECT_SOURCE_DIR}/lib")
